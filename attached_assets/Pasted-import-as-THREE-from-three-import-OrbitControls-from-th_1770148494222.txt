import * as THREE from "three";
import { OrbitControls } from "three/examples/jsm/controls/OrbitControls.js";

interface BambooPole {
  mesh: THREE.Mesh;
  targetHeight: number;
  currentHeight: number;
  ageInYears: number;
  isHarvested: boolean;
  growthFactor: number; // Individual growth rate factor - some poles grow faster than others
}

export class BambooSimulation {
  private scene: THREE.Scene;
  private camera: THREE.PerspectiveCamera;
  private renderer: THREE.WebGLRenderer;
  private controls: OrbitControls;
  private container: HTMLElement;
  private bambooClump: BambooClump;
  private webglAvailable: boolean = true;

  constructor(container: HTMLElement) {
    this.container = container;

    // Scene setup - using white background as requested
    this.scene = new THREE.Scene();
    this.scene.background = new THREE.Color(0xffffff); // White background

    // Camera setup
    this.camera = new THREE.PerspectiveCamera(
      75, // Wider zoom out view to match the screenshot
      container.clientWidth / container.clientHeight,
      0.1,
      1000,
    );
    
    // Position camera further back to see wider view with space around bamboo
    this.camera.position.set(30, 5, 30); // Further back, slightly lower

    // Renderer setup with error handling
    try {
      this.renderer = new THREE.WebGLRenderer({ antialias: true });
      this.renderer.setSize(container.clientWidth, container.clientHeight);
      this.renderer.shadowMap.enabled = true;
      container.appendChild(this.renderer.domElement);
    } catch (error) {
      console.error('WebGL initialization failed:', error);
      throw new Error('WEBGL_NOT_AVAILABLE');
    }

    // Controls
    this.controls = new OrbitControls(this.camera, this.renderer.domElement);
    this.controls.target.set(0, 10, 0); // Target at middle-lower portion of bamboo
    this.controls.enableDamping = true;
    this.controls.dampingFactor = 0.05;
    this.controls.screenSpacePanning = false;
    this.controls.minDistance = 20; // Keep camera from getting too close
    this.controls.maxDistance = 150; // Allow zooming out more
    this.controls.maxPolarAngle = Math.PI / 2;
    
    this.controls.update();

    // Setup lighting and environment
    this.setupLighting();
    this.setupGround();

    // Create the bamboo clump
    this.bambooClump = new BambooClump(new THREE.Vector3(0, 0, 0));
    this.scene.add(this.bambooClump.container);

    // Start animation loop
    this.animate();
  }

  private setupLighting(): void {
    // Stronger ambient light for cleaner look
    const ambientLight = new THREE.AmbientLight(0xffffff, 0.8);
    this.scene.add(ambientLight);

    // Main directional light (sun) - reduced intensity
    const mainLight = new THREE.DirectionalLight(0xffffff, 0.6);
    mainLight.position.set(20, 30, 20);
    mainLight.castShadow = true;
    mainLight.shadow.mapSize.width = 2048;
    mainLight.shadow.mapSize.height = 2048;

    // Configure shadow camera
    mainLight.shadow.camera.left = -30;
    mainLight.shadow.camera.right = 30;
    mainLight.shadow.camera.top = 30;
    mainLight.shadow.camera.bottom = -30;
    mainLight.shadow.camera.near = 0.5;
    mainLight.shadow.camera.far = 100;

    this.scene.add(mainLight);

    // Secondary light from the other side (fill light)
    const fillLight = new THREE.DirectionalLight(0xffffff, 0.2);
    fillLight.position.set(-20, 15, -20);
    this.scene.add(fillLight);
  }

  private setupGround(): void {
    // Clean white ground
    const groundGeometry = new THREE.PlaneGeometry(100, 100);
    // Using MeshBasicMaterial for flat appearance
    const groundMaterial = new THREE.MeshBasicMaterial({
      color: 0xffffff, // Pure white
      side: THREE.DoubleSide,
    });
    const ground = new THREE.Mesh(groundGeometry, groundMaterial);
    ground.rotation.x = -Math.PI / 2;
    this.scene.add(ground);

    // Subtle grid with light gray lines
    const primaryColor = 0xdddddd; // Light gray primary grid lines
    const secondaryColor = 0xf2f2f2; // Very light gray secondary grid lines
    const gridHelper = new THREE.GridHelper(
      100,
      100,
      primaryColor,
      secondaryColor,
    );
    gridHelper.position.y = 0.01; // Slightly above ground
    this.scene.add(gridHelper);

    // Add scale indicators - simple human figure for scale
    this.addScaleIndicator();
  }

  private addScaleIndicator(): void {
    // Create a human figure for scale (1.8m tall)
    const humanGroup = new THREE.Group();

    // Simple human shape
    const bodyGeometry = new THREE.CylinderGeometry(0.2, 0.2, 1.8, 8);
    const bodyMaterial = new THREE.MeshStandardMaterial({ color: 0x333333 });
    const body = new THREE.Mesh(bodyGeometry, bodyMaterial);
    body.position.y = 0.9; // Half height
    body.castShadow = true;
    humanGroup.add(body);

    // Head
    const headGeometry = new THREE.SphereGeometry(0.2, 16, 16);
    const head = new THREE.Mesh(headGeometry, bodyMaterial);
    head.position.y = 1.8;
    head.castShadow = true;
    humanGroup.add(head);

    // Position the human for scale reference
    humanGroup.position.set(3, 0, 3);
    this.scene.add(humanGroup);
  }

  loadTerrain(terrainUrl: string): void {
    // This is a simplified version that doesn't actually load terrain
    // When regional view is requested, we can add a simple map texture here
    console.log("Terrain loading requested:", terrainUrl);
  }
  
  // Camera control methods
  zoomIn(): void {
    // Move camera closer to bamboo
    if (this.controls.getDistance() > this.controls.minDistance + 5) {
      this.camera.position.multiplyScalar(0.8); // Zoom in by 20%
      this.controls.update();
    }
  }
  
  zoomOut(): void {
    // Move camera further from bamboo
    if (this.controls.getDistance() < this.controls.maxDistance - 5) {
      this.camera.position.multiplyScalar(1.2); // Zoom out by 20%
      this.controls.update();
    }
  }
  
  resetCameraPosition(): void {
    // Set camera to default position with good overview of bamboo
    this.camera.position.set(12, 12, 12); // Closer default view
    this.controls.target.set(0, 5, 0); // Focus on middle height of bamboo
    this.controls.update();
  }
  
  // Top-down view
  topView(): void {
    this.camera.position.set(0, 30, 0);
    this.controls.target.set(0, 0, 0);
    this.controls.update();
  }
  
  // Side view
  sideView(): void {
    this.camera.position.set(20, 5, 0);
    this.controls.target.set(0, 5, 0);
    this.controls.update();
  }

  private animate = (): void => {
    requestAnimationFrame(this.animate);

    // Update controls for smooth movement
    this.controls.update();

    // Render the scene
    this.renderer.render(this.scene, this.camera);
  };

  public update(progress: number): {
    height: number;
    poleCount: number;
    carbon: number;
  } {
    return this.bambooClump.update(progress);
  }

  public handleResize(): void {
    // Update camera and renderer on window resize
    this.camera.aspect =
      this.container.clientWidth / this.container.clientHeight;
    this.camera.updateProjectionMatrix();
    this.renderer.setSize(
      this.container.clientWidth,
      this.container.clientHeight,
    );
  }

  public cleanup(): void {
    // Remove the canvas from the DOM
    if (this.renderer.domElement.parentNode) {
      this.renderer.domElement.parentNode.removeChild(this.renderer.domElement);
    }

    // Dispose of geometries and materials
    this.scene.traverse((object) => {
      if (object instanceof THREE.Mesh) {
        if (object.geometry) object.geometry.dispose();

        if (object.material) {
          if (Array.isArray(object.material)) {
            object.material.forEach((material) => material.dispose());
          } else {
            object.material.dispose();
          }
        }
      }
    });

    // Dispose of renderer
    this.renderer.dispose();
  }
}

/**
 * BambooClump class to manage a natural-looking clump of bamboo
 */
class BambooClump {
  public container: THREE.Group;
  private poles: BambooPole[] = [];
  private position: THREE.Vector3;
  private maxPoles: number = 100; // Maximum poles in a mature clump
  private currentPoles: number = 0;
  private harvestedPoles: number = 0;
  private targetHeight: number = 0.5; // Starting height
  private growthProgress: number = 0; // 0 to 1, representing 2025 to 2035
  private rhizomeMesh: THREE.Mesh | null = null;

  constructor(position: THREE.Vector3) {
    this.position = position;
    this.container = new THREE.Group();
    this.container.position.copy(position);

    // Create the rhizome base mound
    this.createRhizomeMound();

    // Initial saplings
    this.addNewPoles(5);
  }

  private createRhizomeMound(): void {
    // Create a mound to represent the rhizome
    const moundGeometry = new THREE.SphereGeometry(1.5, 16, 8);
    moundGeometry.scale(1, 0.3, 1); // Flatten it

    // Cut the bottom half off
    for (let i = 0; i < moundGeometry.attributes.position.count; i++) {
      const y = moundGeometry.attributes.position.getY(i);
      if (y < 0) {
        moundGeometry.attributes.position.setY(i, 0);
      }
    }

    const moundMaterial = new THREE.MeshStandardMaterial({
      color: 0x8b4513, // Brown color
      roughness: 1.0,
    });

    this.rhizomeMesh = new THREE.Mesh(moundGeometry, moundMaterial);
    this.rhizomeMesh.receiveShadow = true;
    this.rhizomeMesh.castShadow = true;
    this.container.add(this.rhizomeMesh);
  }

  // Growth follows a logistic curve (S-curve) for more natural growth
  private calculateHeight(progress: number): number {
    // Logistic growth curve parameters
    // Scale is consistent with the floor grid markings (5m intervals)
    // 10% shorter as requested
    const maxHeight = 2.55; // Maximum visual height (reduced by 10% from 5.0)
    const minHeight = 0.5; // Start with small saplings
    
    const growthRate = 15; // Steepness of the curve - even faster for more dramatic growth
    const midpoint = 0.3; // Point of maximum growth (year 3 = 2028)
    
    // Logistic growth formula - reaches near mature height by year 5 (2030)
    // Add minHeight to ensure we start at the minimum height
    const calculatedHeight = minHeight + (maxHeight - minHeight) / (1 + Math.exp(-growthRate * (progress - midpoint)));
    
    // Log for debugging
    console.log(`Progress: ${progress.toFixed(2)}, Height: ${calculatedHeight.toFixed(2)}m`);
    
    return calculatedHeight;
  }

  private calculatePoleCount(progress: number): number {
    // Start with 5 poles, end with maxPoles (90)
    // Use S-curve for pole count as well for more natural density progression
    const minPoles = 5;
    const growthRate = 15; // Steepness matches height curve
    const midpoint = 0.3; // Same midpoint as height for consistency
    
    // Logistic growth formula for pole count
    const calculatedPoleCount = minPoles + (this.maxPoles - minPoles) / 
      (1 + Math.exp(-growthRate * (progress - midpoint)));
    
    const finalPoleCount = Math.floor(calculatedPoleCount);
    console.log(`Progress: ${progress.toFixed(2)}, Poles: ${finalPoleCount}`);
    
    return finalPoleCount;
  }

  private calculateCarbon(height: number, poleCount: number): number {
    // Carbon calculation based on Mindanao research (87.5 tonnes CO2/ha/year)
    // Adjust formula to reach 87.5 tonnes when at full height (27m) with full poles (90)
    return parseFloat(((height / 27) * (poleCount / 90) * 87.5).toFixed(1));
  }

  private addNewPoles(count: number): void {
    // Get current year based on growth progress
    const currentYear = 2025 + Math.floor(this.growthProgress * 10);
    
    // Track pole positions to avoid intersections
    const existingPositions: {x: number, z: number, radius: number}[] = [];
    
    // Get existing pole positions
    this.poles.forEach(pole => {
      if (pole.mesh && !pole.isHarvested) {
        existingPositions.push({
          x: pole.mesh.position.x,
          z: pole.mesh.position.z,
          radius: pole.mesh.geometry instanceof THREE.CylinderGeometry ? 
                 pole.mesh.geometry.parameters.radiusBottom * 1.2 : 0.1 // Add 20% buffer
        });
      }
    });

    // Counter to prevent infinite loops
    let attemptsCounter = 0;
    const maxAttempts = 100;
    
    for (let i = 0; i < count; i++) {
      let validPosition = false;
      let x = 0, z = 0, thickness = 0;
      
      // Try to find non-intersecting position
      while (!validPosition && attemptsCounter < maxAttempts) {
        attemptsCounter++;
        
        // Keep all poles within a tighter clump - more natural for bamboo
        // With some poles in the center
        let radius;
        
        // About 30% chance to place a pole near the center
        if (Math.random() < 0.3) {
          radius = Math.random() * 0.6; // Central poles (0-0.6 radius)
        } else {
          radius = 0.8 + Math.random() * 1.0; // Outer poles (0.8-1.8 radius)
        }
        
        const angle = Math.random() * Math.PI * 2;
        x = Math.cos(angle) * radius;
        z = Math.sin(angle) * radius;
        
        // Thinner poles to reduce intersections
        // Make central poles thinner for more natural density
        if (radius < 0.6) {
          thickness = 0.06 + Math.random() * 0.04; // Thinner central poles
        } else {
          thickness = 0.12 + Math.random() * 0.06; // Slightly thicker outer poles
        }
        
        // Check if this position intersects with any existing poles
        validPosition = true;
        
        for (const pos of existingPositions) {
          // Calculate distance between pole centers
          const dx = x - pos.x;
          const dz = z - pos.z;
          const distance = Math.sqrt(dx * dx + dz * dz);
          
          // If poles would intersect, reject position
          if (distance < (thickness + pos.radius)) {
            validPosition = false;
            break;
          }
        }
      }
      
      // If we couldn't find a valid position after max attempts, skip this pole
      if (!validPosition) {
        continue;
      }
      
      // Add this pole's position to the list so future poles don't intersect with it
      existingPositions.push({
        x: x,
        z: z,
        radius: thickness * 1.2 // Add 20% buffer
      });

      // Create bamboo pole with significant variation
      // Set full height with random variation - some poles taller, some shorter
      const heightVariation = 30 + Math.random() * 25; // Height varies from 30 to 55 units
      const height = heightVariation;
      
      // Random taper for more natural look
      const topThinning = 0.7 + Math.random() * 0.4; // Top diameter varies from 70% to 110% of base
      
      // Reducing taper to make poles more uniform (closer to left image)
      const poleGeometry = new THREE.CylinderGeometry(
        thickness * topThinning, // Top with random thinning
        thickness * (1.1 + Math.random() * 0.2), // Bottom slightly thicker with variation
        height,
        8,
      );
      // Move cylinder origin to bottom
      poleGeometry.translate(0, height / 2, 0);

      // Using exact color from left reference image (darker green) with more variation
      const poleMaterial = new THREE.MeshStandardMaterial({
        color: new THREE.Color(
          0.02 + Math.random() * 0.02, // Very minimal red with slight variation
          0.15 + Math.random() * 0.1, // Much darker green with more variation
          0.01 + Math.random() * 0.03, // Very minimal blue with slight variation
        ),
        roughness: 0.6 + Math.random() * 0.3,
        metalness: 0.05 + Math.random() * 0.1,
      });

      const pole = new THREE.Mesh(poleGeometry, poleMaterial);
      pole.position.set(x, 0, z);
      pole.castShadow = true;
      pole.receiveShadow = true;

      // Scale to start tiny
      pole.scale.y = 0.01;

      // Minimal tilt variation for more parallel stems like in the reference image
      const tiltAngle = Math.PI * 0.02; // Very small tilt variation
      
      // Almost no outward tilt - bamboo should grow straight up with minimal variation
      // This matches the right image where poles are more parallel
      const distanceFromCenter = Math.sqrt(x * x + z * z);
      const outwardDirection = new THREE.Vector2(x, z).normalize();
      
      // Apply very minimal outward tilt and minimal random variation
      // Keep poles mostly parallel with just slight variation for natural appearance
      pole.rotation.x = (outwardDirection.y * 0.01) + (Math.random() - 0.5) * tiltAngle;
      pole.rotation.z = (-outwardDirection.x * 0.01) + (Math.random() - 0.5) * tiltAngle;

      // Add random growth rate factor (0.8 to 1.2) for each pole - some grow faster, some slower
      const growthRateFactor = 1.2 + Math.random() * 0.4;
      
      const poleObject: BambooPole = {
        mesh: pole,
        targetHeight: height,
        currentHeight: 0.5, // Start small
        ageInYears: 0, // New pole
        isHarvested: false,
        growthFactor: growthRateFactor, // Add individual growth rate factor
      };

      this.poles.push(poleObject);
      this.container.add(pole);
    }

    this.currentPoles += count;
  }

  private harvestMaturePoles(growthProgress: number): void {
    // Only harvest after year 3 (2028)
    if (growthProgress < 0.3) return;

    // Find mature poles (3+ years old)
    const maturePoles = this.poles.filter(
      (pole) => pole.ageInYears >= 3 && !pole.isHarvested,
    );

    // Harvest 20% of mature poles
    const polesToHarvest = Math.ceil(maturePoles.length * 0.2);

    if (polesToHarvest === 0) return;

    // Sort to harvest oldest first
    maturePoles.sort((a, b) => b.ageInYears - a.ageInYears);

    for (let i = 0; i < polesToHarvest && i < maturePoles.length; i++) {
      const pole = maturePoles[i];
      pole.isHarvested = true;

      // Fade out harvested poles
      if (pole.mesh.material instanceof THREE.MeshStandardMaterial) {
        pole.mesh.material.opacity = 0.3;
        pole.mesh.material.transparent = true;
      }

      this.harvestedPoles++;
    }
  }

  public update(growthProgress: number): {
    height: number;
    poleCount: number;
    carbon: number;
  } {
    // Ensure growth progress is always in valid range (0-1)
    this.growthProgress = Math.max(0, Math.min(1, growthProgress));

    // Calculate target height based on valid growth progress
    this.targetHeight = this.calculateHeight(this.growthProgress);
    const targetPoles = this.calculatePoleCount(this.growthProgress);

    // Add new poles if needed (smoothly)
    if (targetPoles > this.currentPoles) {
      this.addNewPoles(targetPoles - this.currentPoles);
    }

    // Limit harvesting to avoid disrupting animation
    // Random check with low probability to smooth out the harvesting
    if (Math.random() < 0.02 && this.growthProgress > 0.5) {
      this.harvestMaturePoles(this.growthProgress);
    }

    // Update existing poles for CONTINUOUS growth with individual variation
    this.poles.forEach((pole) => {
      // Age poles continuously
      pole.ageInYears = this.growthProgress * 10;

      // Skip harvested poles
      if (pole.isHarvested) return;

      // Apply the S-curve height calculation to get the correct scale factor
      // This applies the logistic growth curve for more natural growth
      const normalizedHeight = (this.targetHeight - 0.5) / (4.5 - 0.5); // Normalize to 0-1 range using new max height
      
      // Apply individual growth factor to get varied heights
      // Some poles grow faster, some slower - creates a more natural look
      const individualGrowth = normalizedHeight * (pole.growthFactor || 1.0);
      
      // Clamp the growth between 0.01 and 0.99 for visual stability
      const scaleY = Math.min(0.99, Math.max(0.01, individualGrowth * 0.99 + 0.01));
      
      // Apply the individualized scale to the pole
      pole.mesh.scale.y = scaleY;

      // Set current height for metrics
      pole.currentHeight = pole.targetHeight * scaleY;

      // Update color based on maturity with more variation
      const maturityFactor = Math.min(pole.ageInYears / 5, 1);
      const hue = 0.25 + (0.08 + Math.random() * 0.04) * maturityFactor; // From green to yellow-green with variation
      const saturation = 0.8 - (0.25 + Math.random() * 0.1) * maturityFactor; // Reduce saturation with age with variation
      const lightness = 0.45 + Math.random() * 0.1; // Add variation to lightness

      if (pole.mesh.material instanceof THREE.MeshStandardMaterial) {
        pole.mesh.material.color.setHSL(hue, saturation, lightness);
      }
    });

    // Ensure valid return values (never negative)
    const finalHeight = Math.max(0.5, this.targetHeight);
    const finalPoleCount = Math.max(5, this.currentPoles - this.harvestedPoles);
    const finalCarbon = Math.max(
      0,
      this.calculateCarbon(finalHeight, finalPoleCount),
    );

    return {
      height: finalHeight,
      poleCount: finalPoleCount,
      carbon: finalCarbon,
    };
  }
}
